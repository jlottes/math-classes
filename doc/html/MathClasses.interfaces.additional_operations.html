  }.

Class NatLogSpec {A} N b log `{Equiv A} `{Le A} `{Lt A} `{Zero A} `{One A} `{Plus A} `{Mult A} :=
  { nat_log_proper :>> Morphism (N₊ ⇒ N⁺) log
  ; nat_log_base : b ∊ N₊
  ; nat_log_spec {pw} `{!NatPowSpec N N⁺ pw} x `{x ∊ N₊} : b^(log x) ≤ x < b^(1+log x)
